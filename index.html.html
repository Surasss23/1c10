#!/bin/bash read a read b echo "Enter two numbers:" # Taking user input echo "Enter your choice:" # Input type of operation echo "1. Addition" echo "2. Subtraction" echo "3. Multiplication" echo "4. Division" read ch # Acting on the user input case $ch in 1) res=$(echo "$a + $b" | bc) ;; 2) res=$(echo "$a - $b" | bc) ;; 3) res=$(echo "$a * $b" | bc) ;; 4) res=$(echo "scale=2; $a / $b" | bc) ;; esac # Printing result echo "Result: $res" # Output Examples: # Input: # 5 # 10 # 1 # Result: 15 # -------------------------- # Input: # 10 # 3 # 4 # Result: 3.33 # ---------------- PYTHON SCRIPT ---------------- a = 10 b = 20 c = a + b print("Sequential Execution:") print(f"The sum of {a} and {b} is {c}") # Selection (if-else) print("\nSelection (if-else):") if a > b: print(f"{a} is greater than {b}") else: print(f"{a} is not greater than {b}") # Repetition (for loop) print("\nRepetition (for loop):") for i in range(5): print(f"Iteration {i}") # Repetition (while loop) print("\nRepetition (while loop):") count = 0 while count < 5: print(f"Count is {count}") count += 1
