#include <stdio.h>
#include <unistd.h>
int main() {
    int pipefds[2];
    pipe(pipefds);
    // Create a child process
    int pid = fork();
    if(pid == 0) {
        // Child process 
        char buf[100];
        read(pipefds[0], buf, 100);
        printf("Child received: %s\n", buf);
        close(pipefds[0]);
        close(pipefds[1]);
    } else {
        // Parent process
        char buf[] = "Hello From Parent!";
        write(pipefds[1], buf, sizeof(buf));
        close(pipefds[0]);
        close(pipefds[1]);
    }
    return 0;
}
